def read_file(file_path):
    try:
        with open(file_path, "r", encoding="utf-8") as file:
            return file.read()
    except FileNotFoundError:
        return "文件不存在"


def write_file(file_path, content):
    with open(file_path, "a", encoding="utf-8") as file:
        file.write(content)

2. 定义带成绩校验的 Student 类

class Student:
    def __init__(self, student_id, name, grade="大一", score=0):
        self.student_id = student_id
        self.name = name
        self.grade = grade
        self.score = score

    def update_score(self, new_score):
        if 0 <= new_score <= 100:
            self.score = new_score
        else:
            print("无效成绩")

    def show_info(self):
        print(
            f"学号：{self.student_id}，姓名：{self.name}，"
            f"年级：{self.grade}，成绩：{self.score}"
        )


stu1 = Student("2024001", "张三", "大二")
stu1.update_score(92)
stu1.show_info()
stu1.update_score(105)
stu1.show_info()  # 仍为 92

1. 用继承与多态实现车辆行驶

定义 Vehicle 父类；Car 增加座位数、重写 run() 并实现鸣笛；Bicycle 重写 run()；定义统一函数 vehicle_run(vehicle)。

class Vehicle:
    def __init__(self, brand, color):
        self.brand = brand
        self.color = color

    def run(self):
        print(f"{self.brand}{self.color}车辆正在行驶")


class Car(Vehicle):
    def __init__(self, brand, color, seat_count):
        super().__init__(brand, color)
        self.seat_count = seat_count

    def run(self):
        print(f"{self.brand}{self.color}汽车（{self.seat_count} 座）正在公路行驶")

    def honk(self):
        print("汽车鸣笛：嘀嘀嘀！")


class Bicycle(Vehicle):
    def run(self):
        print(f"{self.brand}{self.color}自行车正在非机动车道行驶")


def vehicle_run(vehicle):
    vehicle.run()


my_car = Car("特斯拉", "红色", 5)
my_bike = Bicycle("捷安特", "黑色")
vehicle_run(my_car)
vehicle_run(my_bike)
my_car.honk()
